library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity EJercicio is
    Port ( a, b, c, d, s: in  STD_LOGIC_VECTOR (0 downto 1);
           z : out  STD_LOGIC_VECTOR (0 downto 1));
end EJercicio;

architecture arq of EJercicio is
begin
process (a,b,c,d,s)
begin
case(s) is
	when "00" => z <= a;
	when "01" => z <= b;
	when "10" => z <= c;
	when others => z <= d;
end case;
end process;
end arq;
